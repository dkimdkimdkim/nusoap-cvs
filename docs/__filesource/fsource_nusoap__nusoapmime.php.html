  69.     function addAttachment($data, $filename = '', $contenttype = 'application/octet-stream', $cid = false) {
    70. 

  70.         if (! $cid) {
    71. 

  71.             $cid = md5(uniqid(time()));
    72. 

  72.         }

  73.  
    74. 

  74.         $info['data'] = $data;

  114.     function getHTTPBody($soapmsg) {
    115. 

  115.         if (count($this->requestAttachments) > 0) {
    116. 

  116.             $params['content_type'] = 'multipart/related';
    117. 

  117.             $mimeMessage =& new Mail_mimePart('', $params);
    118. 

  118.             unset($params);
    119. 

  119.  
    120. 

  120.             $params['content_type'] = 'text/xml';
    121. 

  121.             $params['encoding']     = '8bit';
    122. 

  122.             $params['charset']      = $this->soap_defencoding;
    123. 

  123.             $mimeMessage->addSubpart($soapmsg, $params);
    124. 

  124.             
    125. 

  125.             foreach ($this->requestAttachments as $att) {
    126. 

  126.                 unset($params);
    127. 

  127.  
    128. 

  128.                 $params['content_type'] = $att['contenttype'];
    129. 

  129.                 $params['encoding']     = 'base64';
    130. 

  130.                 $params['disposition']  = 'attachment';
    131. 

  131.                 $params['dfilename']    = $att['filename'];
    132. 

  132.                 $params['cid']          = $att['cid'];
    133. 

  133.  
    134. 

  134.                 if ($att['data'] == '' && $att['filename'] <> '') {
    135. 

  135.                     if ($fd = fopen($att['filename'], 'rb')) {
    136. 

  136.                         $data = fread($fd, filesize($att['filename']));
    137. 

  137.                         fclose($fd);
    138. 

  138.                     } else {
    139. 

  139.                         $data = '';
    140. 

  140.                     }
    141. 

  141.                     $mimeMessage->addSubpart($data, $params);
    142. 

  142.                 } else {
    143. 

  143.                     $mimeMessage->addSubpart($att['data'], $params);
    144. 

  144.                 }
    145. 

  145.             }
    146. 

  146.  
    147. 

  147.             $output = $mimeMessage->encode();

  148.             $mimeHeaders = $output['headers'];
    149. 

  149.     
    150. 

  150.             foreach ($mimeHeaders as $k => $v) {
    151. 

  151.                 $this->debug("MIME header $k: $v");
    152. 

  152.                 if (strtolower($k) == 'content-type') {
    153. 

  153.                     // PHP header() seems to strip leading whitespace starting
    154. 

  154.                     // the second line, so force everything to one line
    155. 

  155.                     $this->mimeContentType = str_replace("\r\n", " ", $v);
    156. 

  156.                 }
    157. 

  157.             }
    158. 

  158.     
    159. 

  159.             return $output['body'];
    160. 

  160.         }
    161. 

  161.  
    162. 

  162.         return parent::getHTTPBody($soapmsg);
    163. 

  163.     }

  196.     /**
    197. 

  197.     * processes SOAP message returned from server
    198. 

  198.     *
    199. 

  199.     * @param    array    $headers    The HTTP headers
    200. 

  200.     * @param    string    $data        unprocessed response data from server

  204.     function parseResponse($headers, $data) {
    205. 

  205.         $this->debug('Entering parseResponse() for payload of length ' . strlen($data) . ' and type of ' . $headers['content-type']);
    206. 

  206.         $this->responseAttachments = array();
    207. 

  207.         if (strstr($headers['content-type'], 'multipart/related')) {
    208. 

  208.             $this->debug('Decode multipart/related');
    209. 

  209.             $input = '';
    210. 

  210.             foreach ($headers as $k => $v) {
    211. 

  211.                 $input .= "$k: $v\r\n";
    212. 

  212.             }
    213. 

  213.             $params['input'] = $input . "\r\n" . $data;
    214. 

  214.             $params['include_bodies'] = true;
    215. 

  215.             $params['decode_bodies'] = true;
    216. 

  216.             $params['decode_headers'] = true;
    217. 

  217.             
    218. 

  218.             $structure = Mail_mimeDecode::decode($params);
    219. 

  219.  
    220. 

  220.             foreach ($structure->parts as $part) {
    221. 

  221.                 if (!isset($part->disposition)) {
    222. 

  222.                     $this->debug('Have root part of type ' . $part->headers['content-type']);
    223. 

  223.                     $return = parent::parseResponse($part->headers, $part->body);
    224. 

  224.                 } else {
    225. 

  225.                     $this->debug('Have an attachment of type ' . $part->headers['content-type']);
    226. 

  226.                     $info['data'] = $part->body;
    227. 

  227.                     $info['filename'] = isset($part->d_parameters['filename']) ? $part->d_parameters['filename'] : '';
    228. 

  228.                     $info['contenttype'] = $part->headers['content-type'];
    229. 

  229.                     $info['cid'] = $part->headers['content-id'];
    230. 

  230.                     $this->responseAttachments[] = $info;
    231. 

  231.                 }
    232. 

  232.             }
    233. 

  233.         
    234. 

  234.             if (isset($return)) {
    235. 

  235.                 return $return;
    236. 

  236.             }
    237. 

  237.             
    238. 

  238.             $this->setError('No root part found in multipart/related content');
    239. 

  239.             return;
    240. 

  240.         }
    241. 

  241.         $this->debug('Not multipart/related');
    242. 

  242.         return parent::parseResponse($headers, $data);
    243. 

  243.     }
